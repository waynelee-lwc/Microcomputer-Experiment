IOY0 EQU 0640H
A0 EQU IOY0+00H
B0 EQU IOY0+02H
C0 EQU IOY0+04H
MODE EQU IOY0+06H
CODE SEGMENT 
	ASSUME CS:CODE
START: 
	MOV DX,MODE
	MOV AL,90H
	OUT DX,AL
LOOP1:
	MOV DX,A0
	IN AL,DX
	CMP AL,0FFH
	JE QUIT
	MOV DX,B0
	OUT DX,AL
	MOV AH,01H
	INT 16H
	JZ LOOP1
QUIT:
	MOV AH,4CH
	INT 21H
CODE ENDS
END START
