CODE	SEGMENT 'CODE'
	ASSUME CS:CODE
START:  MOV AL,90H   ;控制方式字不变
	MOV DX,0646H
	OUT DX,AL
APORT:  MOV DX,0640H
	IN  AL,DX	;读A端口
	AND AL,01H	;与运算判断K0是否为低电平
	JZ B1		;若K0不是低电平，则点亮高4位
	MOV AL,0F0H
	JMP B2
B1:     MOV AL,0FH
B2:     MOV DX,0642H
	OUT DX,AL
	JMP APORT
	MOV AH,4CH  ;返回
	INT 21H
CODE    ENDS
END     START
