CODE SEGMENT
	ASSUME CS:CODE
START:
	CALL SET;设置工作方式
	
	MOV AX, 100
	MOV DX, 0602H
	OUT DX, AL
	MOV AL, AH
	OUT DX, AL

MLOOP:
	JMP MLOOP
	
SET PROC NEAR
	MOV DX, 0606H ;使 计数器0 工作在 方式3
	MOV AL, 36H
	OUT DX, AL
	
	MOV DX, 0606H ;使 计数器1 工作在 方式3
	MOV AL, 76H
	OUT DX, AL
	
	MOV DX, 0600H ;让 计数器0 开始工作，计数初值为 4800H
	MOV AX, 4800H
	OUT DX, AL
	MOV AL, AH
	OUT DX, AL
	
	RET
SET ENDP

CODE ENDS
	END START
